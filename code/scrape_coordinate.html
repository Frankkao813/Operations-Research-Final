Problem statement¶

In order to get the longitude and latitude of a location, one method is to scrape the map data. More specifically, the task is to

  1. use selenium to get to the website desired
  2. type the place desired in the search box
  3. click the search button
  4. switch to the floating block and click the corrdinate button
  5. scrape the information needed and save into Python list

The code segment is adapted from a post in medium.

options = webdriver.ChromeOptions()
options.add_argument("headless")
def get_coordinate(addr):
    # ,options=options
    browser = webdriver.Chrome(executable_path='chromedriver', options=options)
    browser.get("http://www.map.com.tw/")
    # search = browser.find_element_by_id("searchWord")
    search = browser.find_element(By.ID, "searchWord")
    search.clear()
    search.send_keys(addr)
    browser.find_element(By.ID, "btnAddrSearch").click()
    #browser.find_element_by_xpath("/html/body/form/div[10]/div[2]/img[2]").click() 
    time.sleep(5)
    #iframe = browser.find_elements(By.TAG_NAME, "iframe")[1]
    #iframe = browser.find_elements_by_tag_name("iframe")[1]
    #browser.switch_to.frame(iframe)
    browser.switch_to.default_content()
    browser.switch_to.frame(0)
    coor_btn = browser.find_element(By.XPATH, "/html/body/form/div[4]/table/tbody/tr[3]/td/table/tbody/tr/td[2]")
    coor_btn.click()

    coor = browser.find_element(By.XPATH, "/html/body/form/div[5]/table/tbody/tr[2]/td")
    coor = coor.text.strip().split(" ")
    lat = coor[-1].split("")[-1]
    log = coor[0].split("")[-1]
    time.sleep(1)
    browser.quit()
    return (lat, log)

result = []
start_time = time.time()
for i in range(100): # pilot program
    approx_address = df_taipei.iloc[i]["city"] + df_taipei.iloc[i]["district"] + df_taipei.iloc[i]["street"]
    #print(approx_address)
    try:
        coord_return = get_coordinate(approx_address)
        result.append(coord_return)
    except Exception as e:
        print(f"wrong index: {i}, {e}")
        coord_return  = (-1, -1)
        result.append(coord_return)
        continue
    print(f"{i} complete")
end_time = time.time()
time_elasped = end_time - start_time
print(f"time elapsed: {time_elasped}")
